	\item Show that a forest with $n$ vertices and $m$ components has $n-m$ edges.\\
	Suppose the components have $n_1,n_2,n_3,...,n_m$ vertices. The total vertices of the forest is then 
	$n_1+n_2+n_3+...+n_m=n$. The $i$th component is a tree, so it has $n_i-1$ edges by Theorem 4. The total amount of edges in

	\ee

\item Consider a full binary tree $T$ of height $h$.
	\be
	% 10a
	\item How many leaves does $T$ have?\\
	A full binary tree has $2^h$ leaves.
	% 10b
	\item How many vertices does $T$ have?\\
	A full $m$-ary tree has $\frac{m^{h+1}-1}{m-1}$ vertices, so\\
	a full binary tree has $\frac{2^{h+1}-1}{2-1}=2^{h+1}-1$ vertices.
	\ee

	\item Calculate the first few terms of the sequence.\\
	$a_2=\frac{a_1^2-1}{a_0}=\frac{4-1}{1}=3$\\
	$a_3=\frac{a_2^2-1}{a_1}=\frac{9-1}{2}=4$
	% 8b
	\item Using part (a), guess a general formula for $a_n$.\\
	$a_n=n+1$
	% 8c
	\item Prove the guess in part (b).\\
		\begin{itemize}
		\item (Statement) \\
			$a_n=\frac{a_{n-1}^2-1}{a_{n-2}} = n+1$ when $n\geq2$.
		\item (Base) \\
			When $n=2\text{, }a_2=\frac{a_1^2-1}{a_0}=\frac{4-1}{1}=3$ which equals $n+1$ so our base case holds true.
		\item (Induction Hypothesis) \\
			$a_{n-1}=(n-1)+1=n$ for $n\geq2$.\\
			$a_{n-2}=(n-2)+1=n-1$ for $n\geq2$.
		\item (Inductive Step) \\
			\begin{align}
			a_n &= \frac{a_{n-1}^2-1}{a_{n-2}}\\
			    &= \frac{n^2-1}{n-1} & (\text{By Induction Hypothesis}) \\
			    &= \frac{(n-1)(n+1)}{n-1} \\
			    &= n+1
			\end{align}
			Thus, by induction, $a_n=n+1$ for $n\geq2$.
		\end{itemize}
